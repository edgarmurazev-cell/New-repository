<script>
  // --- ДАННЫЕ ИГРОКА ---
  const player = {
    name: "Герой",
    level: 1,
    hp: 30,
    maxHp: 30,
    mp: 10,
    maxMp: 10,
    atk: 4,
    def: 0,
    xp: 0,
    xpToNext: 20,
    gold: 0,
    weaponLevel: 0,
    armorLevel: 0,
  };

  // --- СПИСОК ВРАГОВ ---
  const enemies = [
    { name: "Лесной слизень", maxHp: 15, atk: 3, zone: "Лес новичков", rewardXp: 8, rewardGold: 5 },
    { name: "Дикий волк", maxHp: 24, atk: 5, zone: "Лес новичков", rewardXp: 12, rewardGold: 8 },
    { name: "Разбойник", maxHp: 32, atk: 7, zone: "Окраина деревни", rewardXp: 18, rewardGold: 12 },
    { name: "Теневой маг", maxHp: 40, atk: 9, zone: "Старая башня", rewardXp: 25, rewardGold: 18 },
  ];

  let currentEnemyIndex = 0;
  let enemy = createEnemy(enemies[currentEnemyIndex]);

  // --- ЭЛЕМЕНТЫ DOM ---
  const elPlayerName = document.getElementById("playerName");
  const elPlayerLevel = document.getElementById("playerLevel");
  const elPlayerGold = document.getElementById("playerGold");
  const elHpBar = document.getElementById("hpBar");
  const elMpBar = document.getElementById("mpBar");
  const elXpBar = document.getElementById("xpBar");
  const elHpText = document.getElementById("hpText");
  const elMpText = document.getElementById("mpText");
  const elXpText = document.getElementById("xpText");
  const elDefText = document.getElementById("defText");
  const elAtkText = document.getElementById("atkText");
  const elZoneText = document.getElementById("zoneText");

  const elEnemyName = document.getElementById("enemyName");
  const elEnemyHpText = document.getElementById("enemyHpText");
  const elEnemyInfoText = document.getElementById("enemyInfoText");
  const elEnemyHpBar = document.getElementById("enemyHpBar");

  const elWeaponText = document.getElementById("weaponText");
  const elArmorText = document.getElementById("armorText");

  const elLog = document.getElementById("log");

  const btnAttack = document.getElementById("btnAttack");
  const btnSkill = document.getElementById("btnSkill");
  const btnHeal = document.getElementById("btnHeal");
  const btnNextEnemy = document.getElementById("btnNextEnemy");
  const btnUpgradeWeapon = document.getElementById("btnUpgradeWeapon");
  const btnUpgradeArmor = document.getElementById("btnUpgradeArmor");

  // --- ИНИЦИАЛИЗАЦИЯ ---
  function createEnemy(template) {
    return {
      name: template.name,
      hp: template.maxHp,
      maxHp: template.maxHp,
      atk: template.atk,
      zone: template.zone,
      rewardXp: template.rewardXp,
      rewardGold: template.rewardGold,
    };
  }

  function updateUI() {
    elPlayerName.textContent = player.name;
    elPlayerLevel.textContent = player.level;
    elPlayerGold.textContent = player.gold;

    elHpText.textContent = `HP ${player.hp} / ${player.maxHp}`;
    elMpText.textContent = `MP ${player.mp} / ${player.maxMp}`;
    elXpText.textContent = `XP ${player.xp} / ${player.xpToNext}`;
    elDefText.textContent = `Броня ${player.def}`;
    elAtkText.textContent = `Атака ${player.atk}`;
    elZoneText.textContent = `Локация: ${enemy.zone}`;

    const hpPercent = Math.max(0, (player.hp / player.maxHp) * 100);
    const mpPercent = Math.max(0, (player.mp / player.maxMp) * 100);
    const xpPercent = Math.max(0, (player.xp / player.xpToNext) * 100);

    elHpBar.style.width = hpPercent + "%";
    elMpBar.style.width = mpPercent + "%";
    elXpBar.style.width = xpPercent + "%";

    elEnemyName.textContent = enemy.name;
    elEnemyHpText.textContent = `HP ${enemy.hp} / ${enemy.maxHp}`;
    elEnemyInfoText.textContent = `Атака ${enemy.atk} • Локация: ${enemy.zone}`;
    const enemyHpPercent = Math.max(0, (enemy.hp / enemy.maxHp) * 100);
    elEnemyHpBar.style.width = enemyHpPercent + "%";

    elWeaponText.textContent = `Ржавый меч (+${player.weaponLevel * 2})`;
    elArmorText.textContent = `Тряпичная броня (+${player.armorLevel * 2})`;
  }

  function log(message) {
    const time = new Date().toLocaleTimeString("ru-RU", { hour12: false, hour: "2-digit", minute: "2-digit", second: "2-digit" });
    const div = document.createElement("div");
    div.className = "log-line";
    div.innerHTML = `<span>[${time}]</span> ${message}`;
    elLog.prepend(div);
  }

  function clampPlayer() {
    if (player.hp > player.maxHp) player.hp = player.maxHp;
    if (player.hp < 0) player.hp = 0;
    if (player.mp > player.maxMp) player.mp = player.maxMp;
    if (player.mp < 0) player.mp = 0;
  }

  function playerAttack(isSkill = false) {
    if (enemy.hp <= 0) {
      log("❗ Враг уже повержен. Нажми «Следующий враг».");
      return;
    }

    let damage = player.atk + player.weaponLevel * 2;
    let mpCost = 0;

    if (isSkill) {
      mpCost = 3;
      if (player.mp < mpCost) {
        log("⚠️ Недостаточно маны для навыка.");
        return;
      }
      damage = Math.round(damage * 1.8);
      player.mp -= mpCost;
      log(`✨ Ты используешь навык и наносишь ${damage} урона!`);
    } else {
      log(`⚔️ Ты атакуешь и наносишь ${damage} урона.`);
    }

    enemy.hp -= damage;
    if (enemy.hp <= 0) {
      enemy.hp = 0;
      log(`🏆 Враг «${enemy.name}» повержен! +${enemy.rewardXp} XP, +${enemy.rewardGold} золота.`);
      player.xp += enemy.rewardXp;
      player.gold += enemy.rewardGold;
      checkLevelUp();
    } else {
      enemyTurn();
    }

    clampPlayer();
    updateUI();
  }

  function enemyTurn() {
    if (enemy.hp <= 0) return;
    let damage = enemy.atk - player.def;
    if (damage < 1) damage = 1;
    player.hp -= damage;
    log(`💥 ${enemy.name} наносит тебе ${damage} урона.`);
    if (player.hp <= 0) {
      player.hp = 0;
      log("☠️ Ты пал в бою! Здоровье восстановлено, но часть маны потеряна.");
      player.hp = Math.round(player.maxHp * 0.7);
      player.mp = Math.max(0, player.mp - 2);
    }
  }

  function checkLevelUp() {
    while (player.xp >= player.xpToNext) {
      player.xp -= player.xpToNext;
      player.level++;
      player.maxHp += 6;
      player.maxMp += 2;
      player.atk += 1;
      player.def += 1;
      player.xpToNext = Math.round(player.xpToNext * 1.4);
      player.hp = player.maxHp;
      player.mp = player.maxMp;
      log(`⬆️ Уровень повышен! Теперь ты ${player.level} уровня.`);
    }
  }

  function heal() {
    if (player.hp <= 0) {
      log("⚠️ Ты без сознания. Сначала восстановись после смерти (это происходит автоматически после удара врага).");
      return;
    }
    const healAmount = 12;
    const before = player.hp;
    player.hp += healAmount;
    if (player.hp > player.maxHp) player.hp = player.maxHp;
    const actual = player.hp - before;
    if (actual <= 0) {
      log("🧪 Зелье не подействовало — здоровье и так на максимуме.");
    } else {
      log(`🧪 Ты выпиваешь зелье и восстанавливаешь ${actual} HP.`);
      enemyTurn();
    }
    clampPlayer();
    updateUI();
  }

  function nextEnemy() {
    currentEnemyIndex++;
    if (currentEnemyIndex >= enemies.length) {
      currentEnemyIndex = 0;
      log("🌍 Круг врагов завершён. Начинается новый цикл с усиленными противниками.");
    }
    const template = enemies[currentEnemyIndex];
    const scaled = {
      ...template,
      maxHp: template.maxHp + player.level * 4,
      atk: template.atk + Math.floor(player.level / 2),
    };
    enemy = createEnemy(scaled);
    log(`🎯 Новый враг: ${enemy.name} (локация: ${enemy.zone}).`);
    updateUI();
  }

  function upgradeWeapon() {
    const cost = 10;
    if (player.gold < cost) {
      log("💰 Недостаточно золота для улучшения оружия.");
      return;
    }
    player.gold -= cost;
    player.weaponLevel++;
    log(`🗡️ Оружие улучшено до уровня ${player.weaponLevel}. Атака увеличена.`);
    updateUI();
  }

  function upgradeArmor() {
    const cost = 10;
    if (player.gold < cost) {
      log("💰 Недостаточно золота для улучшения брони.");
      return;
    }
    player.gold -= cost;
    player.armorLevel++;
    player.def += 1;
    log(`🛡️ Броня улучшена до уровня ${player.armorLevel}. Защита увеличена.`);
    updateUI();
  }

  // --- ОБРАБОТЧИКИ КНОПОК ---
  btnAttack.addEventListener("click", () => playerAttack(false));
  btnSkill.addEventListener("click", () => playerAttack(true));
  btnHeal.addEventListener("click", heal);
  btnNextEnemy.addEventListener("click", nextEnemy);
  btnUpgradeWeapon.addEventListener("click", upgradeWeapon);
  btnUpgradeArmor.addEventListener("click", upgradeArmor);

  // Старт
  updateUI();
  log("👋 Добро пожаловать в Mini MMO Arena. Начни с атаки по врагу!");
</script>
